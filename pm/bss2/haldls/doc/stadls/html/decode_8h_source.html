201  words.at(2) = word_type(select(gen));
202  words.at(5) = word_type(select(gen));

207  0, haldls::vx::CHIP_REVISION_STR::CapMemBlockConfig::PrescalePause::max);

209  std::uniform_int_distribution<uintmax_t> lsbs(0, 0xfffff);
